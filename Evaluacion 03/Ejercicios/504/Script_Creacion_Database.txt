DROP DATABASE IF EXISTS ejer504;
CREATE DATABASE ejer504;
USE ejer504;

## Entidad Depto
CREATE TABLE Depto(
numdep decimal(2,0) primary key, 
nomdep varchar(14) default NULL, 
localidad varchar(13) default NULL,
codjefe decimal(4,0) default NULL
);

## Entidad Emp
CREATE TABLE Emp(
numemp decimal(4,0) primary key, 
nomemp varchar(10), 
puesto varchar(10), 
feccont date, 
sal decimal(7,2), 
comision decimal(7,2), 
numdep decimal(2,0),
 FOREIGN KEY (numdep) REFERENCES Depto(numdep)
);
